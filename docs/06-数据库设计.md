# 06. 数据库设计

> **Nova Studio — AI Full Stack Builder**  
> 版本: v1.0 | 日期: 2026-07-16  
> 数据库: PostgreSQL 16

---

## 6.1 ER 关系总览

```
users ──┬── organizations ──┬── projects ──┬── project_files ── file_versions
        │                   │              ├── conversations ── messages
        │                   │              ├── components
        │                   │              ├── theme_configs
        │                   │              ├── agent_flows ── agent_executions
        │                   │              ├── deployments
        │                   │              └── github_repos ── github_syncs
        │                   │
        │                   ├── organization_members
        │                   ├── invitations
        │                   ├── subscriptions
        │                   ├── usage_records
        │                   └── activity_logs
        │
        ├── api_keys
        └── notifications

ai_models (独立表，全局共享)
```

## 6.3 索引策略

```sql
-- 高频查询索引
CREATE INDEX idx_projects_org_id ON projects(org_id);
CREATE INDEX idx_projects_org_slug ON projects(org_id, slug);
CREATE INDEX idx_files_project_path ON project_files(project_id, path);
CREATE INDEX idx_messages_conv_created ON messages(conversation_id, created_at);
CREATE INDEX idx_conversations_project ON conversations(project_id);
CREATE INDEX idx_deployments_project ON deployments(project_id, created_at DESC);
CREATE INDEX idx_usage_org_resource ON usage_records(org_id, resource, recorded_at);
CREATE INDEX idx_activity_org_created ON activity_logs(org_id, created_at DESC);
CREATE INDEX idx_components_project_cat ON components(project_id, category);

-- 全文搜索
CREATE INDEX idx_projects_search ON projects
  USING gin(to_tsvector('english', name || ' ' || coalesce(description,'')));
CREATE INDEX idx_components_search ON components
  USING gin(to_tsvector('english', name || ' ' || coalesce(description,'')));
```

## 6.4 ENUM 类型定义

```sql
-- 组织计划
CREATE TYPE org_plan AS ENUM ('free', 'pro', 'team', 'enterprise');

-- 成员角色
CREATE TYPE member_role AS ENUM ('owner', 'admin', 'member', 'viewer');

-- 成员状态
CREATE TYPE member_status AS ENUM ('active', 'invited', 'suspended');

-- 项目类型
CREATE TYPE project_type AS ENUM (
  'website', 'saas', 'crm', 'dashboard',
  'landing_page', 'ai_agent', 'blog', 'ecommerce', 'internal_tool'
);

-- 项目状态
CREATE TYPE project_status AS ENUM ('draft', 'active', 'archived');

-- 消息角色
CREATE TYPE message_role AS ENUM ('user', 'assistant', 'system', 'tool');

-- AI 提供商
CREATE TYPE ai_provider AS ENUM ('openai', 'anthropic', 'google', 'groq', 'ollama', 'mistral');

-- 部署平台
CREATE TYPE deploy_platform AS ENUM ('vercel', 'netlify', 'custom');

-- 部署状态
CREATE TYPE deploy_status AS ENUM ('queued', 'building', 'deploying', 'live', 'failed', 'cancelled');

-- 部署环境
CREATE TYPE deploy_environment AS ENUM ('preview', 'production');

-- Agent 触发类型
CREATE TYPE agent_trigger_type AS ENUM ('manual', 'webhook', 'schedule', 'event');

-- Agent 状态
CREATE TYPE agent_status AS ENUM ('draft', 'active', 'paused');

-- Agent 执行状态
CREATE TYPE agent_execution_status AS ENUM ('pending', 'running', 'completed', 'failed', 'cancelled');

-- 用量资源类型
CREATE TYPE usage_resource AS ENUM ('ai_tokens', 'ai_messages', 'build_minutes', 'deployments');

-- 邀请状态
CREATE TYPE invite_status AS ENUM ('pending', 'accepted', 'declined', 'expired');

-- 订阅状态
CREATE TYPE subscription_status AS ENUM ('active', 'canceled', 'past_due', 'trialing');

-- GitHub 同步方向
CREATE TYPE github_sync_direction AS ENUM ('push', 'pull');

-- GitHub 同步状态
CREATE TYPE github_sync_status AS ENUM ('pending', 'synced', 'failed');
```
